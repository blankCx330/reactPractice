### 2.2 必须能回答的面试问题

#### 基础问题（必问）

1. **什么是虚拟列表？为什么需要它？**
   ```
   答案要点：
   - 虚拟列表是一种只渲染可视区域内元素的技术
   - 解决大量数据渲染导致的DOM节点过多、内存占用高、滚动卡顿问题
   - 适用于：长列表、聊天记录、商品列表、新闻Feed等场景
   ```

2. **虚拟列表的核心原理是什么？**
   ```
   答案要点：
   - 计算可视区域能容纳的元素数量 = containerHeight / itemHeight
   - 根据滚动位置计算起始索引 startIndex = scrollTop / itemHeight
   - 只渲染 startIndex 到 endIndex 范围内的元素
   - 使用 transform 或 absolute 定位偏移已滚动的高度
   - 复用DOM节点，避免频繁创建销毁
   ```

3. **定高和不定高虚拟列表有什么区别？**
   ```
   答案要点：
   定高（FixedSizeList）：
   - 每个元素高度固定，计算简单
   - startIndex = scrollTop / itemHeight
   - 偏移量 = startIndex * itemHeight
   
   不定高（VariableSizeList）：
   - 元素高度不固定，需要预估或测量
   - 需要缓存每个元素的实际高度
   - 偏移量需要累加之前所有元素的高度
   - 实现复杂度高，但更通用
   ```

#### 进阶问题（高频追问）

4. **如何处理动态高度的列表项？**
   ```
   答案要点：
   方案一：预估高度 + 实测修正
   - 初始使用预估高度进行定位
   - 元素渲染后获取实际高度
   - 更新缓存，重新计算偏移量
   
   方案二：二分查找定位
   - 维护高度前缀和数组
   - 使用二分查找快速定位startIndex
   - 时间复杂度O(log n)
   ```

5. **快速滚动时出现白屏怎么解决？**
   ```
   答案要点：
   - 增加缓冲区（overscan），上下多渲染几条
   - 使用占位骨架屏，提升视觉体验
   - 使用requestAnimationFrame优化滚动事件
   - 预渲染下一屏内容
   ```

#### 4.3.2 API封装

```typescript
// src/api/articleApi.ts

import type { TrendingRepo, TrendingParams } from '@/types/article'

// GitHub Trending API（公开API，无跨域，无需代理）
const BASE_URL = 'https://githubtrending.lessx.xyz'

export async function fetchTrendingRepos(
  params: TrendingParams = {}
): Promise<TrendingRepo[]> {
  const { since = 'daily', language, spoken_language_code } = params
  
  // 构建查询参数
  const searchParams = new URLSearchParams()
  searchParams.set('since', since)
  if (language) searchParams.set('language', language)
  if (spoken_language_code) searchParams.set('spoken_language_code', spoken_language_code)
  
  const url = `${BASE_URL}/trending?${searchParams.toString()}`
  
  try {
    const response = await fetch(url)
    
    if (!response.ok) {
      throw new Error(`请求失败: ${response.status}`)
    }
    
    const data: TrendingRepo[] = await response.json()
    return data
  } catch (error) {
    console.error('获取GitHub Trending数据失败:', error)
    throw error
  }
}

// 获取指定语言的热门仓库
export async function fetchJavascriptTrending(): Promise<TrendingRepo[]> {
  return fetchTrendingRepos({ language: 'javascript', since: 'daily' })
}

// 获取指定时间范围的热门仓库
export async function fetchTrendingBySince(
  since: 'daily' | 'weekly' | 'monthly'
): Promise<TrendingRepo[]> {
  return fetchTrendingRepos({ since })
}
```

> **API使用说明**：
> 
> | 参数 | 类型 | 说明 |
> |------|------|------|
> | `since` | `daily` / `weekly` / `monthly` | 时间范围，默认 `daily` |
> | `language` | string | 编程语言过滤，如 `javascript`、`python`、`typescript` |
> | `spoken_language_code` | string | 自然语言过滤，如 `zh`、`en` |
> 
> **注意事项**：
> 1. API为公开服务，建议缓存数据避免频繁请求

#### 4.3.3 React Query Hook

```typescript
// src/components/ArticleList/useArticleList.ts

import { useQuery } from '@tanstack/react-query'
import { fetchTrendingRepos } from '@/api/articleApi'

interface UseTrendingReposOptions {
  since?: 'daily' | 'weekly' | 'monthly'
  language?: string
  enabled?: boolean
}

export function useTrendingRepos(options: UseTrendingReposOptions = {}) {
  const { since = 'daily', language, enabled = true } = options

  return useQuery({
    queryKey: ['trendingRepos', since, language],
    queryFn: () => fetchTrendingRepos({ since, language }),
    staleTime: 30 * 60 * 1000,  // 30分钟缓存（数据更新不频繁）
    gcTime: 60 * 60 * 1000,     // 1小时后清理
    enabled,
    retry: 2,
  })
}

// 带分页的Hook（用于无限滚动模拟）
export function useTrendingReposPaginated(pageSize: number = 20) {
  const { data, isLoading, error, ...rest } = useTrendingRepos()
  
  // 模拟分页数据
  const paginatedData = data ? data.slice(0, pageSize) : []
  const hasMore = data ? data.length > pageSize : false
  
  return {
    data: paginatedData,
    allData: data,
    isLoading,
    error,
    hasMore,
    ...rest,
  }
}
```

#### 4.3.4 虚拟列表组件

```tsx
// src/components/ArticleList/index.tsx

import { useRef, useState, useCallback } from 'react'
import { FixedSizeList as List } from 'react-window'
import { useTrendingRepos } from './useArticleList'
import RepoItem from './RepoItem'

export default function TrendingList() {
  const [since, setSince] = useState<'daily' | 'weekly' | 'monthly'>('daily')
  const { data: repos, isLoading, error } = useTrendingRepos({ since })
  
  const listRef = useRef<List>(null)

  // 时间范围切换
  const handleSinceChange = (newSince: 'daily' | 'weekly' | 'monthly') => {
    setSince(newSince)
    listRef.current?.scrollTo(0)
  }

  // 列表项渲染
  const Row = useCallback(
    ({ index, style }: { index: number; style: React.CSSProperties }) => {
      if (!repos || index >= repos.length) {
        return (
          <div style={style} className="flex items-center justify-center">
            <span className="text-gray-400">加载中...</span>
          </div>
        )
      }

      const repo = repos[index]
      return <RepoItem repo={repo} style={style} />
    },
    [repos]
  )

  if (isLoading) {
    return <div className="flex justify-center py-10">加载中...</div>
  }

  if (error) {
    return (
      <div className="flex flex-col items-center justify-center py-10 gap-2">
        <span className="text-red-500">加载失败，请稍后重试</span>
        <button 
          onClick={() => window.location.reload()}
          className="px-4 py-2 bg-blue-500 text-white rounded"
        >
          重试
        </button>
      </div>
    )
  }

  return (
    <div className="h-screen bg-gray-50 dark:bg-gray-900 flex flex-col">
      {/* 顶部筛选栏 */}
      <div className="flex gap-2 p-4 bg-white dark:bg-gray-800 border-b">
        {(['daily', 'weekly', 'monthly'] as const).map((item) => (
          <button
            key={item}
            onClick={() => handleSinceChange(item)}
            className={`px-3 py-1 rounded ${
              since === item
                ? 'bg-blue-500 text-white'
                : 'bg-gray-100 dark:bg-gray-700 text-gray-700 dark:text-gray-300'
            }`}
          >
            {item === 'daily' ? '今日' : item === 'weekly' ? '本周' : '本月'}
          </button>
        ))}
      </div>

      {/* 虚拟列表 */}
      {repos && repos.length > 0 && (
        <List
          ref={listRef}
          height={window.innerHeight - 60}
          itemCount={repos.length}
          itemSize={120}
          width="100%"
        >
          {Row}
        </List>
      )}
    </div>
  )
}
```

```tsx
              <img
                key={index}
                src={builder.avatar}
                alt="contributor"
                className="w-5 h-5 rounded-full border border-gray-200 dark:border-gray-600"
              />
            ))}
            {repo.builders.length > 5 && (
              <span className="text-xs text-gray-400 ml-1">
                +{repo.builders.length - 5}
              </span>
            )}
          </div>
        )}
      </div>
    </a>
  )
}

function formatNumber(numStr: string): string {
  const num = parseInt(numStr.replace(/,/g, ''), 10)
  if (isNaN(num)) return numStr
  
  if (num >= 10000) {
    return (num / 10000).toFixed(1) + 'w'
  }
  if (num >= 1000) {
    return (num / 1000).toFixed(1) + 'k'
  }
  return numStr
}
```

## ✍️ 七、手写虚拟列表参考

### 7.1 最简实现（定高）

```tsx
import { useState, useRef, useCallback } from 'react'

interface VirtualListProps<T> {
  items: T[]
  itemHeight: number
  containerHeight: number
  renderItem: (item: T, index: number) => React.ReactNode
  overscan?: number
}

export function VirtualList<T>({
  items,
  itemHeight,
  containerHeight,
  renderItem,
  overscan = 3,
}: VirtualListProps<T>) {
  const [scrollTop, setScrollTop] = useState(0)
  const containerRef = useRef<HTMLDivElement>(null)

  // 计算可视区域内的元素范围
  const startIndex = Math.max(0, Math.floor(scrollTop / itemHeight) - overscan)
  const endIndex = Math.min(
    items.length - 1,
    Math.ceil((scrollTop + containerHeight) / itemHeight) + overscan
  )

  // 可视区域内的元素
  const visibleItems = items.slice(startIndex, endIndex + 1)

  // 总高度（撑开滚动区域）
  const totalHeight = items.length * itemHeight

  // 偏移量（让元素显示在正确位置）
  const offsetY = startIndex * itemHeight

  // 滚动事件处理
  const handleScroll = useCallback((e: React.UIEvent<HTMLDivElement>) => {
    setScrollTop(e.currentTarget.scrollTop)
  }, [])

  return (
    <div
      ref={containerRef}
      style={{ height: containerHeight, overflow: 'auto' }}
      onScroll={handleScroll}
    >
      {/* 撑开滚动区域 */}
      <div style={{ height: totalHeight, position: 'relative' }}>
        {/* 可视元素 */}
        <div style={{ transform: `translateY(${offsetY}px)` }}>
          {visibleItems.map((item, index) => (
            <div
              key={startIndex + index}
              style={{ height: itemHeight }}
            >
              {renderItem(item, startIndex + index)}
            </div>
          ))}
        </div>
      </div>
    </div>
  )
}
```
