```tsx
// SevenTempChart.tsx - 温度曲线组件
export default function SevenTempChart({ chartData }: Props) {
  const theme = useThemeStore(state => state.theme)
  const isDark = theme === 'dark'

  const option = {
    title: { text: title, textStyle: { color: isDark ? '#e5e7eb' : '#1f2937' } },
    tooltip: {
      trigger: 'axis',
      backgroundColor: isDark ? 'rgba(30,30,30,0.9)' : 'rgba(255,255,255,0.95)',
    },
    legend: { top: 35, textStyle: { color: isDark ? '#ffffff' : '#000000' } },
    series: data.map(d => ({
      name: d.name,
      data: d.data,
      color: d.color,
      type: 'line',
      smooth: true,
      markLine: { data: [{ type: 'average', name: '平均' }] },
    })),
  }
  return <EChartsReact option={option} notMerge={true} />
}
```
