```tsx
// useThemeStore.tsx - 主题切换功能
export const useThemeStore = create<ThemeStore>()(
  persist(
    set => ({
      theme: 'light',
      toggleTheme: () => {
        set(state => ({
          theme: state.theme === 'light' ? 'dark' : 'light',
        }))
      },
      setTheme: theme => set({ theme }),
    }),
    { name: 'theme-storage' }
  )
)
```
