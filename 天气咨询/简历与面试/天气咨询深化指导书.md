```tsx
// 每输入一个字符就触发请求
const {data: getCityLocation} = useQuery({
    queryKey: ['searchedCityLocation', inputCityName],  // inputCityName 每次变化都触发
    enabled: inputCityName.length > 1
})
```

**修改方案**：
```tsx

// 使用防抖
const [debouncedInput] = useDebouncedValue(inputCityName, 300)

    queryKey: ['searchedCityLocation', debouncedInput],  // 使用防抖后的值
    enabled: debouncedInput.length > 1
})
```

##### 任务 1.2.4：添加 Error Boundary

**新增文件**：`src/components/ErrorBoundary.tsx`

```tsx
import { Component, ErrorInfo, ReactNode } from 'react'

interface Props {
  children: ReactNode
  fallback?: ReactNode
}

interface State {
  hasError: boolean
  error?: Error
}

export class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('Error caught by boundary:', error, errorInfo)
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback || (
        <div className="error-boundary">
          <h2>出错了</h2>
          <p>{this.state.error?.message}</p>
          <button onClick={() => window.location.reload()}>刷新页面</button>
        </div>
      )
    }
    return this.props.children
  }
}
```

##### 任务 2.2.2：创建 Store

**新增文件**：`src/store/useAppStore.ts`

```tsx
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

interface FavoriteCity {
  id: string
  name: string
  lon: number
  lat: number
}

interface AppState {
  // 状态
  favoriteCities: FavoriteCity[]
  isDarkMode: boolean
  
  // 操作
  addFavoriteCity: (city: FavoriteCity) => void
  removeFavoriteCity: (id: string) => void
  toggleDarkMode: () => void
}

export const useAppStore = create<AppState>()(
  persist(
    (set) => ({
      favoriteCities: [],
      isDarkMode: false,
      
      addFavoriteCity: (city) =>
        set((state) => ({
          favoriteCities: [...state.favoriteCities, city]
        })),
      
      removeFavoriteCity: (id) =>
        set((state) => ({
          favoriteCities: state.favoriteCities.filter((c) => c.id !== id)
        })),
      
      toggleDarkMode: () =>
        set((state) => ({
          isDarkMode: !state.isDarkMode
        }))
    }),
    {
      name: 'weather-app-storage'  // localStorage key
    }
  )
)
```

```tsx
import ReactECharts from 'echarts-for-react'
import type { WeatherDaily } from '../types/qweather'

interface Props {
  daily: WeatherDaily[]
}

export function TemperatureChart({ daily }: Props) {
  const option = {
    title: { text: '7日温度趋势', textStyle: { color: '#fff' } },
    xAxis: {
      type: 'category',
      data: daily.map(d => d.fxDate.slice(5))  // MM-DD
    },
    yAxis: { type: 'value', name: '温度(°C)' },
    series: [
      {
        name: '最高温度',
        type: 'line',
        data: daily.map(d => Number(d.tempMax)),
        smooth: true,
        itemStyle: { color: '#ff6b6b' }
      },
      {
        name: '最低温度',
        type: 'line',
        data: daily.map(d => Number(d.tempMin)),
        smooth: true,
        itemStyle: { color: '#4ecdc4' }
      }
    ],
    legend: { top: 30, textStyle: { color: '#fff' } },
    tooltip: { trigger: 'axis' }
  }

  return <ReactECharts option={option} style={{ height: 300 }} />
}
```
